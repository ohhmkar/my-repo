```py
class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9 + 7
        # Base case when n = 1, possible ABA type = 6 and ABC = 6
        a,b = 6,6
        for i in range(2,n+1):
            a1 = (3*a+2*b)%MOD
            # Transition from ABA to next line can be ABA/BAB or ABA/BAC, so 3 of first type and 2              of second type, sum becomes new number of ABA types
            b1 = (2*(a+b))%MOD
            # Transition from ABC to next line can be ABC/CAB or ABC/BAB both can be of 2 types
            a,b = a1,b1
        return (a+b)%MOD
```
